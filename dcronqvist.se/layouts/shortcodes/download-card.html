{{- $file := .Get "file" -}} {{- $text := .Get "text" -}} {{- $fileInfo := os.Stat (path.Join "static" $file) -}} {{- $fileSize := "" -}} {{- if $fileInfo -}} {{- $bytes := $fileInfo.Size -}} {{- if ge $bytes 1073741824 -}} {{- $fileSize = printf "(%.1f GB)" (div (float $bytes) 1073741824.0) -}} {{- else if ge $bytes 1048576 -}} {{- $fileSize = printf "(%.1f MB)" (div (float $bytes) 1048576.0) -}} {{- else if ge $bytes 1024 -}} {{- $fileSize = printf "(%.1f KB)" (div (float $bytes) 1024.0) -}} {{- else -}} {{- $fileSize = printf "(%d B)" $bytes -}} {{- end -}} {{- end -}} {{ $text }} {{ $fileSize }}
